import { Component, OnInit } from '@angular/core'; import { ActivatedRoute, Router } from '@angular/router'; import { FormsModule } from '@angular/forms'; import { CommonModule } from '@angular/common'; @Component({ selector: 'app-setup-password', standalone: true, imports: [CommonModule, FormsModule], templateUrl: './setup-password.html', styleUrls: ['./setup-password.scss'] }) export class SetupPasswordComponent implements OnInit { password = ''; confirmPassword = ''; token: string | null = ''; isSubmitting = false; constructor(private route: ActivatedRoute, private router: Router) {} ngOnInit(): void { // Extrae el token de la URL (ej: ?token=abc123) this.token = this.route.snapshot.queryParamMap.get('token'); if (!this.token) { alert('Enlace de activación inválido.'); this.router.navigate(['/login']); } } savePassword(): void { if (this.password !== this.confirmPassword) return; this.isSubmitting = true; console.log('Enviando nueva contraseña para el token:', this.token); // Simulación de guardado (Aquí conectarías con tu servicio de backend) setTimeout(() => { alert('¡Cuenta activada con éxito!'); this.isSubmitting = false; this.router.navigate(['/login']); }, 1500); } }
